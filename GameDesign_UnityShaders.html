Intro:
Shaders allow game developers to add detail and unique look and feel to individual art assets models and the overall look of the game. New shaders are as simple to add to Unity as any other script.

STEPS TO IMPLEMENT A SHADER SCRIPT:
 

1. In Unity, Rightclick in the Project panel to make a new Unlit shader. Name it to match the script Shader name.

Example below: Name the shaderscript "SectionPlane".





2. Open the ShaderScript in your text editor and replace all the shadercode with the script.

Example: See script below


3. Save and click on Unity to compile.




4. Rightclick in the Project panel to make a new Material.





5. In the Inspector, set the shader to the name of your ShaderScript (in this case, "IppokratisBournellis/SectionPlane" )
6. Drag the Material onto a 3D Object in a scene, like a sphere.






7. Select the material in the Project panel and in the Inspector play with the x/y/z and w values to see the cross section work.




 
Sample shader script, to get a cross-section of 3D geometry in-game:
"Section Plane" by Ippokratis Bournellis
(Source: post #45 in this thread)

SectionPlane.cs:
Shader IppokratisBournellis/SectionPlane
{
      Properties
      {
            _MainTex ("Texture", 2D) = "white" {}
            _section ("Section plane (x angle, y angle, z angle, w displacement)", vector) = (0,0,0,0.15)
            _Color ("Section Color", Color) = (1,1,1,0)
      }

      SubShader
      {
            Tags { "RenderType" = "Opaque" }
            Cull Off

            CGPROGRAM
            #pragma surface surf Lambert

            struct Input
            {
                  float2 uv_MainTex;
                  float3 worldPos;
                  float3 viewDir;
                  float3 worldNormal;
            };

            sampler2D _MainTex;
            sampler2D _BumpMap;
            float4 _section;
            fixed4 _Color;

            void surf (Input IN, inout SurfaceOutput o)
            {
                  float toClip = _section.x * 0.1 * IN.worldPos.x +
                              _section.y * 0.1 * IN.worldPos.y +
                              _section.z * 0.1 * IN.worldPos.z +
                              _section.w;

            clip( toClip);

            float fd = dot( IN.viewDir, IN.worldNormal);

            if (fd.x > 0)
            {
                  o.Albedo = tex2D (_MainTex, IN.uv_MainTex).rgb;
                  return;
            }

            o.Emission = _Color;
            }
            ENDCG
      }
      Fallback "Diffuse"
}
